def ecdf(data):
    """Compute ECDF for a one-dimensional array of measurements."""
    # Number of data points: n
    n = len(data)

    # x-data for the ECDF: x
    x = np.sort(data)

    # y-data for the ECDF: y
    y = np.arange(1, n+1) / n

    return x, y

Testando hipteses Nossa hiptese nula aqui que a distribuio de terremotos por ms uniformemente aleatria. Para testar esta suposio, usaremos a diferena entre a varincia das distribuies como nossa estatstica de teste e geraremos um grande nmero de valores a partir de amostras bootstrap. Isso ser realizado pelas seguintes etapas:

Combine os dados empricos com dados uniformemente distribudos do mesmo tamanho. Permute os dados combinados com substituio e gere duas novas amostras. Este um exemplo de bootstrap. Repita o processo um grande nmero de vezes e calcule a estatstica de teste para cada amostra. Este um conjunto de rplicas de bootstrap. Trace um histograma do conjunto e calcule o valor p de nossa estatstica de teste original. Se o valor p for pequeno, a varincia das duas distribuies estatisticamente significativamente diferente. Caso contrrio, no podemos descartar a hiptese nula.

Comeamos definindo algumas funes personalizadas para nossa tarefa.

def permutation_sample(data1, data2):
    """Generate a permutation sample from two data sets."""

    # Concatenate the data sets: data
    data = np.concatenate((data1, data2))

    # Permute the concatenated array: permuted_data
    permuted_data = np.random.permutation(data)

    # Split the permuted array into two: perm_sample_1, perm_sample_2
    perm_sample_1 = permuted_data[:len(data1)]
    perm_sample_2 = permuted_data[len(data1):]

    return perm_sample_1, perm_sample_2

def draw_perm_reps(data_1, data_2, func, size=1):
    """Generate multiple permutation replicates."""

    # Initialize array of replicates: perm_replicates
    perm_replicates = np.empty(size)

    for i in range(size):
        # Generate permutation sample
        perm_sample_1, perm_sample_2 = permutation_sample(data_1, data_2)

        # Compute the test statistic
        perm_replicates[i] = func(perm_sample_1, perm_sample_2)

    return perm_replicates

def diff_of_mean(data_1, data_2):
      """Difference in variance of two arrays."""
  
      # The difference of means of data_1, data_2: diff
      diff = np.mean(data_1) - np.mean(data_2)
  
      return diff

def diff_of_var(data_1, data_2):
      """Difference in variance of two arrays."""
  
      # The difference of means of data_1, data_2: diff
      diff = np.var(data_1) - np.var(data_2)
  
      return diff

def pearson_r(x, y):
    """Compute Pearson correlation coefficient between two arrays."""
    # Compute correlation matrix: corr_mat
    corr_mat = np.corrcoef(x, y)

    # Return entry [0,1]
    return corr_mat[0,1]
